Масиви (продовження)

Винесемо фрагмент коду для виведення усіх результатів з обчисленням максимального результату в функцію printAndGetHighScore, а у функцію getBestResults занесемо фрагмент для обчислення кращих результатів що рівні максимуму.

Нехай, ми маємо ще один масив

var costs = [25, 27, 25, ... ];

Потрібно визначити зразок з максимальним результатом у якому витрати мінімальні. Масив costs повторює структуру масиву scores тобто витрати на виробництво кожного зразку зберігаються в елементі з тим самим індексом що і результат цього зразку. Таким масиви називають паралельними.

Напишемо псевдокод програми:

  оголосити змінну cost та присвоїти їй 100

        надати index значення змінній i

  RETURN index
}

Закріплення знань

Написати програми:

1) Знайти перший від'ємний елемент масиву.

var array = [23, -32, 12, 14, -1, 4, -6];
var k = 0;

console.log(array);

for (var i = 0; i < array.length; i++) {
  if (array[i] < 0) {
    console.log("Перший від'ємний елемент масиву = " + array[i]);
    break;
  }
}

2) Знайти мінімальний елемент з непарним індексом.

var array = [4,5,2,-7,1];
var min = 0;

for (var i = 0; i < array.length; i++) {
  if (array[i] % 2 != 0) {
    if (min > array[i] || min == 0) {
      min = array[i];
    }
  }
}

if (min == 0) {
  console.log("В масиві немає непарних елементів");
} else {
  console.log(min + " є мінімальним непарним елементом в масиві");
}

3) Вставка елемента в масив.

var array = [1,2,3,4,5];
console.log(array);
var k = 3;
var n = 0;

array.length = array.length + 1;

for (var i = array.length-1; i > k; i--) {
  array[i] = array[i-1];
}

array[k] = n;

console.log(array);

4) Вилучення елемента з масива.

var array = [1,2,3,4,5];
console.log(array);
var k = 3;

for (var i = k; i < array.length - 1; i++) {
  array[i] = array[i+1];
}

array.length = array.length - 1;

console.log(array);

5) Зсув елементів масиву (<-, на 1).

var array = [1,2,3,4,5];
var firstElement = array[0];

for (var i = 1; i < array.length; i++) {
  array[i-1] = array[i];
}

array[array.length-1] = firstElement;

console.log(array);
