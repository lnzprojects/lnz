Огляд методів для рядків

Властивість length повертає кількість символів у рядку (зручно використовувати для перебору символів у рядку).

var text = "hello";
console.log(text.length);

Метод charAt() отримує ціле число від 0 до довжини рядку (мінус 1) і повертає рядок з одним символом, що знаходиться в заданій позиції. Рядок можна рознлядати як масив, елементами якого є окремі символи, а індекси починаються з 0. Якщо переданий індекс більше або дорівнює довжині рядку, повертаєтсья порожній рядок.

var input = "jenny@wickedlysmart.com";

for (var i = 0; i < input.length; i++) {
  if (input.charAt(i) == "@") {
    console.log("There's an @ sign at index" + i);
  }
}

Метод indexOf() отримує рядок-аргумент і повертає індекс першого символу першого входження аргумента в своєму рядку.

var phrase = "The cat in the hat";
var index = phrase.indexOf("cat");
console.log("There's a cat sitting at index " + index);

Також можна передати другий аргумент - індекс початкової позиції пошуку.

var phrase = "The cat in the hat";
var index = phrase.indexOf("cat", 5);
console.log("There's a cat sitting at index " + index);

Якщо рядок знайти не вдалось, метод повертає -1.

Метод lastIndexOf() знаходить останнє входження підрядка в рядку.

Метод substring() отримує два індекса, після чого витягує і повертає рядок між ними (не включаючи 2й індекс).

var data = "name|phone|address";
var val = data.substring(5, 10);
console.log("Substring is " + val);

Другий індекс можна не вказувати, в цьому випадку substring дістане рядок, який починається з першого індекса і продовжується до кінця рядку.

var data = "name|phone|address";
var val = data.substring(5);
console.log("Substring is now " + val);

Метод split() отримує символ-обмежувач і розбиває рядок на частини по позиції обмежувача.

var data = "name|phone|address";
var vals = data.split("|");
console.log("Split array is ", vals);

В цьому випадку масив vals не перетворюється в рядок перед виведенням в консоль, тому в console.log передається два аргумента, розділених комою.

Метод trim() вилучає пропуски з обоx кінців рядку.

var text = "   hello  ";
var newtext = text.trim();
console.log("New text is " + newtext);

1) Визначити кількість слів у рядку.

var text = "   hello my  friend  ";
text = text.trim();
var i = 0;

while (i < text.length - 1) {
  if (text[i] == " " && text[i+1] == " ") {
    text = text.substring(0, i) + text.substring(i+1, text.length);
    i--;
  }
  i++;
}

text = text.split(" ");
var n = text.length;
console.log("Кількість слів у рядку = " + n);

2) Зробити усі перші букви слів великими (toUpperCase).

var text = "   hello  my   friend  ";
text = text[0].toUpperCase() + text.substring(1, text.length);

for (var i = 1; i < text.length-1; i++) {
  if (text[i] !== " " && text[i-1] == " ") {
    text = text.substring(0, i) + text[i].toUpperCase() + text.substring(i+1, text.length);
  }
}

console.log(text);

3) Визначити довжину найдовшого слова.

var text = "   hello my  friend  ";
text = text.trim();
var i = 0;

while (i < text.length - 1) {
  if (text[i] == " " && text[i+1] == " ") {
    text = text.substring(0, i) + text.substring(i+1, text.length);
    i--;
  }
  i++;
}

text = text.split(" ");
var max = text[0];
var nmax = 0;

for (var i = 0; i < text.length; i++) {
  if (text[i].length > nmax) {
    max = text[i];
    nmax = text[i].length;
  }
}

console.log("Найдовше слово - " + max);

4) Написати програму, яка повертає ім'я файла (підрядок після останнього символа "/") з повного шляху до файлу.

var text = "user/documents/project";
var newText = text.substring(text.lastIndexOf("/")+1, text.length);
console.log(newText);

5) Написати програму, яка визначає, приймати номер телефону чи ні (приймати, якщо номер складається з семи цифр від 0 до 9 і після перших трьох цифр є дефіс).

var number = "033-1949";
var ok = false;
var k = 0;

if (number.length == 8) {
  if (number.charAt(3) == "-") {
    ok = true;
    for (var i = 0; i < number.length; i++) {
      if (!isNaN(number[i])) {
        k++;
      }
    }
  }
}

if (k == 7 && ok == true) {
  console.log("Телефон прийнятий!");
} else {
  console.log("Ви ввели неправильний номер!");
}
